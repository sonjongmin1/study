# 제이쿼리

- 자바스크립트의 라이브러리

### 사용방법

- 다운로드 https://jquery.com/download/ -> 프로젝트 파일에 다른이름 저장 -> script src 링크연결

### 기본형태

```js
$(document).ready(function () {
  //실행문
  $("선택자").click(() => {
    $("선택자").css("속성이름", "속성값");
  });
});
```

# 활용 예제1

```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
    <!-- js 라이브러리 -->
    <script src="jquery-3.7.1.min.js"></script>
    <script src="https://code.jquery.com/jquery-3.7.1.js"></script>
  </head>
  <body>
    <h1 id="title">제이쿼리 배워보자</h1>
    <button id="btn">스타일 변경</button>
  </body>

  <script>
    // $(선택자).실행명령();
    $(document).ready(function () {
      $("#btn").click(function () {
        $("#title").css("color", "red").css("font-size", "30px");
      });
    });
  </script>
</html>
```

# 활용 예제2

```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
    <!-- js 라이브러리 -->
    <script src="jquery-3.7.1.min.js"></script>
    <script src="https://code.jquery.com/jquery-3.7.1.js"></script>
  </head>
  <body>
    <div class="profile">
      <h2 id="pname">홍길동</h2>
      <p id="pd">웹개발자</p>
    </div>
    <div class="ed-di">
      <input type="text" id="edinput" placeholder="이름수정..." />
      <button id="ed-name">이름 수정</button>
    </div>
  </body>

  <script>
    $(document).ready(function () {
      $("#ed-name").click(function () {
        let edinput = $("#edinput").val();
        if (edinput.trim()) {
          $("#pname").text(edinput);
        } else {
          alert("수정할 이름을 입력하세요.");
        }
      });
    });

    $("#edinput").keyup(function () {
      $("#pd").html(`<b>실시간 수정하고 있습니다. ${$(this).val()}`);
    });
  </script>
</html>
```

# 활용예제 5

```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
    <script src="jquery-3.7.1.min.js"></script>
    <script src="https://code.jquery.com/jquery-3.7.1.js"></script>
    <style>
      input {
        height: 30px;
      }

      .hoverStyle {
        padding: 20px;
        border: 1px solid black;
      }
    </style>
  </head>
  <body>
    <input type="text" placeholder="텍스트 입력" id="changeText" />
    <button id="changeTextBtn">텍스트 변경</button>
    <div class="hoverStyle">여기에 마우스를 올려보세요.</div>
    <div id="titleHBox">
      <h1 class="titleH">게시물 1내용</h1>
      <button id="rmv1">게시물 삭제</button>
      <button id="rcv1">내용 지우기</button>
    </div>

    <div id="titleHBox2">
      <h1 class="titleH2">게시물 2내용</h1>
      <button id="rmv2">게시물 삭제</button>
      <button id="rcv2">내용 지우기</button>
    </div>

    <script>
      $(document).ready(function () {
        $("#changeTextBtn").click(function () {
          let changeText = $("#changeText").val();
          if (changeText.trim()) {
            $(".hoverStyle").text(changeText);
          }
        });

        $(".hoverStyle").hover(
          function () {
            $(this).css("background-color", "yellow");
          },
          function () {
            $(this).css("background-color", "#fff");
          }
        );

        $("#rmv1").click(function () {
          $("#titleHBox").remove();
        });

        $("#rcv1").click(function () {
          $(".titleH").empty();
        });

        $("#rmv2").click(function () {
          $("#titleHBox2").remove();
        });

        $("#rcv2").click(function () {
          $(".titleH2").empty();
        });
      });
    </script>
  </body>
</html>
```

# 활용예제 6

```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
    <script src="jquery-3.7.1.min.js"></script>
    <script src="https://code.jquery.com/jquery-3.7.1.js"></script>
    <style>
      #btn {
        width: 100px;
        height: 100px;
        border: 1px solid black;
        text-align: center;
        align-content: center;
      }
    </style>
  </head>
  <body>
    <div id="btn">비활성화</div>

    <script>
      let flag = false;
      $(document).ready(function () {
        $("#btn").click(function () {
          if (!flag) {
            $("#btn").css("background-color", "blue");
            $("#btn").css("color", "#fff");
            $("#btn").text("활성화");
            flag = true;
          } else {
            $("#btn").css("background-color", "#fff");
            $("#btn").css("color", "black");
            $("#btn").text("비활성화");
            flag = false;
          }
        });
      });
    </script>
  </body>
</html>
```
